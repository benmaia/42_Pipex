# Colors #
GREEN		=	\033[0;32m
YELLOW		=	\033[0;33m
RED			=	\033[0;31m
RESET       =   \033[0m

NAME = pipex

LIBFT_DIR	=	libft
SRC			=	srcs/main.c
OBJ			=	$(SRC:srcs/%.c=$(OBJ_DIR)/%.o)
INC_DIR		=	inc
BUILD_DIR	=	./build
BIN_DIR		=	$(BUILD_DIR)/bin
OBJ_DIR		=	$(BUILD_DIR)/objects
BIN			:=	$(BIN_DIR)/$(NAME)

CC = gcc

CFLAGS = -Wall -Wextra -Werror -fsanitize=address -I$(INC_DIR) -I$(LIBFT_DIR)

all: $(NAME)

$(NAME): $(BIN)

$(BIN):	$(OBJ) | $(LIBFT) $(BIN_DIR)
		@make -s -C $(LIBFT_DIR)
		@$(CC) $(CFLAGS) -o $(@) $(^) -L$(LIBFT_DIR) -lft

$(OBJ_DIR)/%.o:	srcs/%.c | $(BUILD_DIR)
	$(CC) $(CFLAGS) -c $(^) -o $(@)

.SILENT = $(OBJ)

$(BUILD_DIR):
		mkdir -p $(BUILD_DIR)
		mkdir -p $(OBJ_DIR) $(BIN_DIR)

clean:
	@make clean -s -C $(LIBFT_DIR)
	@rm -rf $(OBJ)
	@#echo "$(YELLOW)[    Objects have been removed!  ]$(RESET)"

fclean: clean
	@make fclean -s -C $(LIBFT_DIR)
	@rm -rf build
	@#echo "$(RED)[     Pipex has been removed!    ]$(RESET)"

re: fclean all

exe: all
	@./build/bin/pipex test.txt "ls" "cat" test2.txt

.PHONY: all clean fclean re
